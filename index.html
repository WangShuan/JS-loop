a++與++a的差別

var a =10
var b=a++ +6
// a=11,b=16
++在後面的時候就要先a=10,b=10+6,a++,a=11

var c=10
var d=++c +6
// c=11,d=17
++在前面的時候就要先c++,c=11,d=11+6
